   // 수정버전
   function letterCapitalize(str) {
     var capitalizeStr = ''

     for (var i = 0; i < str.length; i++) {
       if ((str[i - 1] === undefined || str[i - 1] === ' ') && str[i] !== ' ') {
         capitalizeStr += str[i].toUpperCase()
       } else {
         capitalizeStr += str[i]
       }
     }

     return capitalizeStr
   }
